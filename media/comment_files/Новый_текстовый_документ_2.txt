  <script>
document.getElementById("submit-comment-file").addEventListener("click", function() {
  var commentText = document.getElementById("comment-textarea").value;
  var fileInput = document.getElementById("file-input");
  var fileList = fileInput.files;
  var files = [];
  
  // Проверяем, есть ли выбранный файл
  if (fileList.length > 0) {
    for (var i = 0; i < fileList.length; i++) {
      var file = fileList[i];
      files.push({
        name: file.name,
        url: URL.createObjectURL(file)
      });
    }
  }

  // Создаем новый комментарий
  var commentContainer = document.createElement("div");
  commentContainer.className = "comment";
  commentContainer.textContent = commentText;

  // Прикрепляем файлы к комментарию (если они были выбраны)
  if (files.length > 0) {
    var fileListElement = document.createElement("ul");
    fileListElement.className = "file-list";
    files.forEach(function(file) {
      var listItem = document.createElement("li");
      var fileLink = document.createElement("a");
      fileLink.className = "file-link";
      fileLink.href = file.url; // URL файла
      fileLink.textContent = file.name;
      listItem.appendChild(fileLink);
      fileListElement.appendChild(listItem);
    });
    commentContainer.appendChild(fileListElement);
  }

  // Добавляем комментарий к списку комментариев
  document.getElementById("comments-list").appendChild(commentContainer);

  // Очищаем поля ввода
  document.getElementById("comment-textarea").value = "";
  fileInput.value = "";
});
  </script>
